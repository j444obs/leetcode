## 时间复杂度：O(2n)

## 代码：

function fib(n){
	if(n < 0){
		return -1;
	}else if(n < 2 && n >= 0){
		return n;
	}else{
		return fib(n-1) + fib(n-2)
	}
}
```

## 时间复杂度：O(n)

function fib(n){
	let arr=new Array(n+1).fill(null);  //新建一维数组
	//初始值
	arr[0] = 0;
	arr[1] = 1;
	//数组元素间关系式
	for(let i=2; i<=n; i++){
		arr[i] = arr[i-1] + arr[i-2];
	}
	return arr[n];
}

```
